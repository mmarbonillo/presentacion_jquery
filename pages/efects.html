Efectos Básicos

    x.hide(y)        //Esconde el elemento x a una velocidad de y milisegundos
    x.show(y)        //Muestra el elemento x a una velocidad de y milisegundos
    x.toggle(y)      //Intercala entre hide y show

    <head>
        <script>
            $('#ex1button').click(function(){
                $('#felixex1').show();
            });
            $('#ex1button2').click(function(){
                $('#felixex1').hide();
            });
            $('#ex1button3').click(function(){
                $('#felixex1').toggle();
            });
        </script>
    </head>
    <body>
        <img id="felixex1" src="../images/Captura.JPG" width="500px" style="display: none;">
        <input id="ex1button" class="example1button" type="button" value="show()">
        <input id="ex1button2" class="example1button" type="button" value="hide()">
        <input id="ex1button3" class="example1button" type="button" value="toggle()">

    </body>

Efectos de Fade

    x.fadeIn(y)          //Hace aparecer el elemento x usando la opacidad a una velocidad de y milisegundos
    x.fadeOut(y)         //Hace desaparecer el elemento x usando la opacidad a una velocidad de y milisegundos
    x.fadeTo(y, z)       //Cambia la opacidad (z) del elemento x a una velocidad de y milisegundos
    x.fadeToggle()       //Intercala entre fadeIn y fadeOut

    <head>
        <script>
            $('#ex2button1').click(function(){
                $('#profesex2').fadeIn(1000);
            });
            $('#ex2button2').click(function(){
                $('#profesex2').fadeOut(2000);
            });
            $('#ex2button3').click(function(){
                $('#profesex2').fadeTo(2000, 0.3);
            });
            $('#ex2button4').click(function(){
                $('#profesex2').fadeToggle(1500);
            });
        </script>
    </head>
    <body>
        <div id="imgex2">
            <img id="profesex2" src="../images/profes.jpg" width="500px" style="display: none;">
        </div>
        <input id="ex2button1" class="example2button" type="button" value="fadeIn()">
        <input id="ex2button2" class="example2button" type="button" value="fadeOut()">
        <input id="ex2button3" class="example2button" type="button" value="fadeTo()">
        <input id="ex2button4" class="example2button" type="button" value="fadeToggle()">
        <input id="examplecode2" type="button" class="example2button" value="Ver código">
    </body>
