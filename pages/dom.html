Manipulación Del Árbol DOM

¿Qué es el Árbol DOM?

Es la representacíon gráfica en forma de árbol de la estructura de una página.

¿Qué Podemos Hacer?

Hay muchísimas funciones de manipulación del DOM. A continuación veremos una lista de todas las que disponemos y ejemplos de las más relevantes.

    <script type="text/javascript">

        x.addClass('classname')          //Añade una nueva clase a un elemento
        x.hasClass('classname')          //Comprueba si el elemento x tiene o no la clase 'classname'
        x.toggleClass()
        x.before('content')              //Añade contenido antes de cada etiqueta x
        x.after('content')               //Añade contenido despues de cada etiqueta x
        $('content').insertAfter(x)      //Inserta el contenido especificado despues de cada elemento x
        $('content').insertBefore(x)     //Inserta el contenido especificado antes de cada elemento x
        x.prepend('content')             //Añade contenido al principio de la etiqueta x
        x.prependTo('tag')               //Añade el contenido x al final de la etiqueta 'tag'
        x.append('content')              //Añade contenido al final de la etiqueta x
        x.appendTo('tag')                //Añade el contenido x al final de la etiqueta 'tag'
        $('content').replaceAll(x)       //Remplaza el elemento(s) x por el contenido especificado
        x.replaceWith('content')         //Remplaza el elemento(s) x por el contenido especificado y devuelve
                                         //el número de elementos que ha eliminado
        x.text()                         //Devuelve una cadena de texto con todo el contenido de la página
        x.text('content')                //Sustituye el contenido del elemento(s) x por el especificado.
                                         //No reconoce etiquetas HTML
        x.empty()                        //Vacía (literalmente) la etiqueta x
        x.remove()                       //Elimina el elemento(s) x así como todo el código jQuery asociados a estos
        x.detach()                       //Elimina el elemento(s) x dejando el código jQuery asociado
        x.attr('attrname'{, 'value'})    //Recupera el valor de un atributo o añade/remplaza un atributo a un elemento
        x.clone()                        //Clona el elemento x
        x.css('property'{, 'value'})     //Recupera el valor de una propiedad css o añade una propiedad css
        x.height({'value'})              //Devuelve o asigna la altura del elemento x
        x.width({'value'})               //Devuelve o asigna la anchura del elemento x
        x.html()                         //Devuleve el contenido HTML del elemento x
        x.html('HTMLcontent')            //Añade el contenido HTML a cada uno de los elementos x
        x.innerHeight()                  //Devuelve el alto INTERIOR de un elemento (incluye padding pero no borde)
        x.innerWidth()                   //Devuelve el ancho INTERIOR de un elemento (incluye padding pero no borde)
        x.outerHeigth()                  //Devuelve el alto total del elemento x (incluye margen, borde y padding)
        x.outerWidth()                   //Devuelve el ancho total del elemento x (incluye margen, borde y padding)
        x.prop('propertyname')           //Devuelve el valor de la propiedad especificada del elemento x
        x.prop('propertyname', 'value')  //Asigna la propiedad y su valor a todos los elementos x
        x.removeAttr('attrname')         //Elimina el atributo especificado del elemento(s) x
        x.removeProp('propertyname')     //Elimina la propiedad especificada del elemento(s) x
        x.removeClass('class1 class2')   //Elimina una o varias clases del elemento(s) x
        x.wrap('content')                //Añade el padre 'content' al elemento(s) x
        x.wrapAll()
        x.wrapInner()
        x.unwrap()

    </script>
